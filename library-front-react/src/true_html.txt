import React, { useState } from "react";
import { BrowserRouter as Router, Routes, Route, Link } from "react-router-dom";

import logo from "./logo.svg";
import "./App.css";

import LoginForm from "./components/LoginForm";
import RegisterForm from "./components/RegisterForm";
import SearchBooks from "./components/SearchBooks";

function App() {
  const [token, setToken] = useState(localStorage.getItem("token"));

  return (
    <Router>
      <div className="app-container">
        <header className="App-header">
          <img src={logo} className="App-logo" alt="Logo" />
          <h1 className="App-title">ðŸ“š BibliothÃ¨que</h1>

          {!token && (
            <nav>
              {/* Use Link for client-side navigation */}
              <Link to="/search">
                <button className="btn-search">Search Books</button>
              </Link>
            </nav>
          )}
        </header>

        <main className="App-main">
          <Routes>
            {!token ? (
              <>
                <Route
                  path="/"
                  element={
                    <>
                      <section className="auth-section">
                        <div className="login-form">
                          <h2>Connexion</h2>
                          <LoginForm onLogin={setToken} />
                        </div>
                        <div className="register-form">
                          <h2>Inscription</h2>
                          <RegisterForm />
                        </div>
                      </section>
                    </>
                  }
                />
                {/* Redirect all unknown paths to "/" for unauthenticated users */}
                <Route path="*" element={<LoginForm onLogin={setToken} />} />
              </>
            ) : (
              <>
                <Route
                  path="/search"
                  element={
                    <section className="search-section">
                      <h2>Recherche de Livres</h2>
                      <SearchBooks />
                    </section>
                  }
                />
                {/* Default route when logged in: can redirect to /search or show welcome */}
                <Route
                  path="/"
                  element={
                    <section>
                      <h2>Bienvenue !</h2>
                      <p>Utilisez le bouton "Search Books" pour rechercher des livres.</p>
                    </section>
                  }
                />
              </>
            )}
          </Routes>
        </main>
      </div>
    </Router>
  );
}

export default App;
